# 정적 클래스 멤버 함수 호출

```csharp
class C
{
    public static void F();

    public void G()
    {
        // 1. 클래스 내부에서 호출
        F();
    }
}

// 2. 클래스 외부에서 호출, public 으로 노출되어야 가능하다
C.G();

var c = new C();
c.G(); // 3. 인스턴스 함수처럼 호출하는 허용하지 않는다
       //    허용하면 GetC().G(); 이런 모습이 나올 수 있는데.. G는 C를 필요로 하지 않기 때문에 GetC() 부분이 생략되어도 상관없어 '보이기' 때문이다.
```
